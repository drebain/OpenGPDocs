55  void mouse_button(GLFWwindow *window, int button, int action, int mods) {
56  if (button == GLFW_MOUSE_BUTTON_LEFT && action == GLFW_PRESS) {
57  old_rotation_ = rotation_;
58  double x_i, y_i;
59  glfwGetCursorPos(window, &x_i, &y_i);
60  const float x = 2.0f * static_cast<float>(x_i) / _width - 1.0f;
61  const float y = 1.0f - 2.0f * static_cast<float>(y_i) / _height;
62  trackball_.BeginDrag(x, y);
63  }
64 
65  current_state_ = mods;
66  }

68  void mouse_move(GLFWwindow *window, double x, double y, double old_x, double old_y) {
69  bool view_matrices_changed = false;
70 
71  bool left_down = glfwGetMouseButton(window, GLFW_MOUSE_BUTTON_LEFT) == GLFW_PRESS;
72  bool middle_down = glfwGetMouseButton(window, GLFW_MOUSE_BUTTON_MIDDLE) == GLFW_PRESS;
73  bool right_down = glfwGetMouseButton(window, GLFW_MOUSE_BUTTON_RIGHT) == GLFW_PRESS;
74 
75  // Maya style controls.
76  if ( left_down && current_state_ == GLFW_MOD_NONE ) {
77  const float x_f = 2.0f * static_cast<float>(x) / _width - 1.0f;
78  const float y_f = 1.0f - 2.0f * static_cast<float>(y) / _height;
79  trackball_.Drag(x_f, y_f);
80  rotation_ = trackball_.incremental_rotation() * old_rotation_;
81  view_matrices_changed = true;
82  }
83 
84  const float dx = x - old_x;
85  const float dy = y - old_y;
86 
87  // Pan
88  if ( middle_down || (left_down && (current_state_ == GLFW_MOD_SHIFT || current_state_ == GLFW_MOD_CONTROL)) ) {
89  const float scale = 0.05f;
90  translation_ *= Eigen::Affine3f(Eigen::Translation3f(scale * dx, -scale * dy, 0.0f)).matrix();
91  view_matrices_changed = true;
92  }
93 
94  // Zoom
95  if ( right_down || (left_down && (current_state_ == GLFW_MOD_ALT || current_state_ == GLFW_MOD_SUPER)) ) {
96  const float scale = 0.05f;
97  translation_ *= Eigen::Affine3f(Eigen::Translation3f(0.0f, 0.0f, scale * dy)).matrix();
98  view_matrices_changed = true;
99  }
100 
101  if (view_matrices_changed) {
102  update_matrices();
103  }
104  }

117  update_matrices();

120  void hook_trackball_callbacks( void (*update_matrices)(Eigen::Matrix4f), void (*update_projection_matrix)() ){
122  _update_matrices = update_matrices;

124 
125  _update_projection_matrix = update_projection_matrix;
126  assert(_update_projection_matrix!=NULL);
127 
128  translation_ = Eigen::Matrix4f::Identity();
129  rotation_ = Eigen::Matrix4f::Identity();
130  old_rotation_ = Eigen::Matrix4f::Identity();
131  scale_ = Eigen::Matrix4f::Identity();

133  glfwSetKeyCallback(window, keyboard);
134  glfwSetMouseButtonCallback(window, mouse_button);
135  glfwSetCursorPosCallback(window, mouse_pos);
136  glfwSetScrollCallback(window, mouse_wheel);
