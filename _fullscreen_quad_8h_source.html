16 class FullscreenQuad : public GPUMesh {
17 private:
18 
19  const char* vshader = R"GLSL(
20  #version 330 core
21 
22  in vec3 vposition;
23  out vec2 uv;
24 
25  void main(){
26  gl_Position = vec4(vposition, 1);
27  uv = vec2((vposition.x + 1) / 2, (vposition.y + 1) / 2);
28  }
29  )GLSL";
30 
31  const char* fshader = R"GLSL(
32  #version 330 core
33 
34  uniform sampler2D tex;
35  uniform int grayscale;
36 
37  in vec2 uv;
38  out vec4 fcolor;
39 
40  void main(){
41  vec4 full_color = texture(tex, uv);
42  float red = full_color.r;
43  fcolor = (1 - grayscale) * full_color + grayscale * vec4(red, red, red, 1);
44  }
45  )GLSL";
46 
47  Shader default_shader;
48 
49 public:
50 
51  FullscreenQuad() {
52 
53  default_shader.add_vshader_from_source(vshader);
54  default_shader.add_fshader_from_source(fshader);
55  default_shader.link();
56 
57  std::vector<Vec3> vpoint;
58  vpoint.push_back(Vec3(1, 1, 0));
59  vpoint.push_back(Vec3(-1, 1, 0));
60  vpoint.push_back(Vec3(1, -1, 0));
61  vpoint.push_back(Vec3(-1, -1, 0));
62 
63  set_vpoint(vpoint);
64 
65  std::vector<unsigned int> triangles;
66  triangles.push_back(0);
67  triangles.push_back(3);
68  triangles.push_back(1);
69  triangles.push_back(0);
70  triangles.push_back(2);
71  triangles.push_back(3);
72 
73  set_triangles(triangles);
74 
75  default_shader.bind();
76 
77  set_attributes(default_shader);
78  default_shader.set_uniform("tex", 0);
79 
80  default_shader.unbind();
81 
82  set_mode(GL_TRIANGLES);
83 
84  }
85 
86  virtual ~FullscreenQuad() {}
87 
88  void draw_texture(GenericTexture &texture) {
89  draw_texture(texture, default_shader);
90  }

92  void draw_texture(GenericTexture &texture, Shader &shader) {
93 
94  shader.bind();
95 
96  if (texture.get_format() == GL_RED) {
97  shader.set_uniform("grayscale", 1);
98  } else {
99  shader.set_uniform("grayscale", 0);
100  }
101 
102  glActiveTexture(GL_TEXTURE0); // is this needed?
103  texture.bind();
104 
105  draw();
106 
107  texture.unbind();
108 
109  shader.unbind();
110 
111  }
112 
113 };
