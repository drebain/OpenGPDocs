14 inline Vec2 pill_tangent(const Vec4 &s0_3D, const Vec4 &s1_3D) {
15 
17 
18  Scalar dist_3D = (s1_3D.block<3, 1>(0, 0) - s0_3D.block<3, 1>(0, 0)).norm();
19 
20  Vec3 s0(0, 0, s0_3D(3));
21  Vec3 s1(dist_3D, 0, s1_3D(3));
22 
23  Vec3 a = s1 - s0;
24  Scalar a2 = (a(0)*a(0)) + (a(1)*a(1));
25  Scalar ad = std::sqrt(a2);
26  Vec2 ab = a.block<2, 1>(0, 0);
27  Vec2 an = ab / ad;
28 
29  Scalar beta = ab.dot(Vec2(std::sqrt((ad*ad)-(a(2)*a(2))), a(2))) / (ad * ad);
30  Scalar alpha = std::sqrt(1 - (beta * beta));
31 
32  return Vec2(beta, alpha);
33 
34 }
