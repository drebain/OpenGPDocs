348  void delete_vertex(Vertex v) {
349 
351 
352  if (vdeleted[v]) return;
353 
354  std::vector<Sphere> spheres_to_delete;
355  std::vector<Edge> edges_to_delete;
356  std::vector<Face> faces_to_delete;
357 
358  for (auto s : spheres()) {
359  if (Vertex(sconn[s]) == v) {
360  spheres_to_delete.push_back(s);
361  }
362  }
363 
364  for (auto e : edges()) {
365  if (Vertex(econn[e](0)) == v || Vertex(econn[e](1)) == v) {
366  edges_to_delete.push_back(e);
367  }
368  }
369 
370  for (auto f : faces()) {
371  if (Vertex(fconn[f](0)) == v || Vertex(fconn[f](1)) == v || Vertex(fconn[f](2)) == v) {
372  faces_to_delete.push_back(f);
373  }
374  }
375 
376  for (auto s : spheres_to_delete) {
377  delete_sphere(s);
378  }
379 
380  for (auto e : edges_to_delete) {
381  delete_edge(e);
382  }
383 
384  for (auto f : faces_to_delete) {
385  delete_face(f);
386  }
387 
388  vdeleted[v] = true;
389  deleted_vertices++;
390  has_garbage = true;
391 
392  }

626  bool read_text(const std::string& text) {
627 
628  std::regex vert_regex(R"regex(v\s+([\d\-\.]+)\s+([\d\-\.]+)\s+([\d\-\.]+)\s+([\d\-\.]+))regex");
629 
630  std::sregex_iterator verts_begin(text.begin(), text.end(), vert_regex);
631  std::sregex_iterator verts_end;
632 
633  for (auto i = verts_begin;i != verts_end;++i) {
634 
635  auto match = *i;
636 
637  vprops.push_back();
638  auto it = vertices_end();
639  vpoint[*(--it)] = Point(std::stof(match[1]), std::stof(match[2]), std::stof(match[3]), std::stof(match[4]));
640 
641  }
642 
643  std::regex sphere_regex(R"regex(s\s+([\d]+))regex");
644  std::sregex_iterator spheres_begin(text.begin(), text.end(), sphere_regex);
645  std::sregex_iterator spheres_end;
646 
647  for (auto i = spheres_begin;i != spheres_end;++i) {
648 
649  auto match = *i;
650 
651  sprops.push_back();
652  auto it = this->spheres_end();
653  sconn[*(--it)] = std::stoi(match[1]);
654 
655  }
656 
657  std::regex edge_regex(R"regex(p\s+([\d]+)\s+([\d]+))regex");
658  std::sregex_iterator edges_begin(text.begin(), text.end(), edge_regex);
659  std::sregex_iterator edges_end;
660 
661  for (auto i = edges_begin;i != edges_end;++i) {
662 
663  auto match = *i;
664 
665  eprops.push_back();
666  auto it = this->edges_end();
667  econn[*(--it)] = EdgeConnectivity(std::stoi(match[1]), std::stoi(match[2]));
668 
669  }
670 
671  std::regex face_regex(R"regex(w\s+([\d]+)\s+([\d]+)\s+([\d]+))regex");

673  std::sregex_iterator faces_end;

676

678

749  void garbage_collection() {
750 
752 
753  int i, i0, i1,
754  nV(vertices_size()),
755  nS(spheres_size()),
756  nE(edges_size()),
757  nF(faces_size());
758 
759  VertexProperty<Vertex> vmap = add_vertex_property<Vertex>("v:garbage-collection");
760 
761  for (i = 0;i < nV;i++) {
762  vmap[Vertex(i)] = Vertex(i);
763  }
764 
765  // remove deleted vertices
766  if (nV > 0) {
767  i0=0; i1=nV-1;
768 
769  while (1) {
770  // find first deleted and last un-deleted
771  while (!vdeleted[Vertex(i0)] && i0 < i1) ++i0;
772  while ( vdeleted[Vertex(i1)] && i0 < i1) --i1;
773  if (i0 >= i1) break;
774 
775  // swap
776  vprops.swap(i0, i1);
777  };
778 
779  // remember new size
780  nV = vdeleted[Vertex(i0)] ? i0 : i0+1;
781  }
782 
783  VertexProperty<Vertex> vmap_reverse = add_vertex_property<Vertex>("v:garbage-collection-2");
784 
785  for (i = 0;i < nV;i++) {
786 
787  Vertex vnew(i);
788  Vertex vold = vmap[vnew];
789 
790  vmap_reverse[vold] = vnew;
791 
792  }
793 
794  // remove deleted spheres
795  if (nS > 0) {
796  i0=0; i1=nS-1;
797 
798  while (1) {
799  // find first deleted and last un-deleted
800  while (!sdeleted[Sphere(i0)] && i0 < i1) ++i0;
801  while ( sdeleted[Sphere(i1)] && i0 < i1) --i1;
802  if (i0 >= i1) break;
803 
804  // swap
805  sprops.swap(i0, i1);
806  };
807 
808  // remember new size
809  nS = sdeleted[Sphere(i0)] ? i0 : i0+1;
810  }
811 
812  // remove deleted edges
813  if (nE > 0) {
814  i0=0; i1=nE-1;
815 
816  while (1) {
817  // find first deleted and last un-deleted
818  while (!edeleted[Edge(i0)] && i0 < i1) ++i0;
819  while ( edeleted[Edge(i1)] && i0 < i1) --i1;
820  if (i0 >= i1) break;
821 
822  // swap
823  eprops.swap(i0, i1);
824  };
825 
826  // remember new size
827  nE = edeleted[Edge(i0)] ? i0 : i0+1;
828  }
829 
830  // remove deleted faces
831  if (nF > 0) {
832  i0=0; i1=nF-1;
833 
834  while (1) {
835  // find first deleted and last un-deleted
836  while (!fdeleted[Face(i0)] && i0 < i1) ++i0;
837  while ( fdeleted[Face(i1)] && i0 < i1) --i1;
838  if (i0 >= i1) break;
839 
840  // swap
841  fprops.swap(i0, i1);
842  };
843 
844  // remember new size
845  nF = fdeleted[Face(i0)] ? i0 : i0+1;
846  }
847 
848  for (i = 0;i < nV;i++) {
849 
850  Sphere s(i);
851 
852  sconn[s] = vmap_reverse[Vertex(sconn[s])].idx();
853 
854  }
855 
856  for (i = 0;i < nE;i++) {
857 
858  Edge e(i);
859 
860  econn[e](0) = vmap_reverse[Vertex(econn[e](0))].idx();
861  econn[e](1) = vmap_reverse[Vertex(econn[e](1))].idx();
862 
863  }
864 
865  for (i = 0;i < nF;i++) {
866 
867  Face f(i);
868 
869  fconn[f](0) = vmap_reverse[Vertex(fconn[f](0))].idx();
