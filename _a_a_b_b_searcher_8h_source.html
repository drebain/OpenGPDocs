19 template <typename VertexMatrix, typename FaceMatrix>
20 class AABBSearcher{
22  typedef CGAL::Simple_cartesian<typename VertexMatrix::Scalar> K;
23  typedef typename K::FT FT;
24  typedef typename K::Point_3 Point_3;
25  typedef typename K::Triangle_3 Triangle;
26  typedef typename std::vector<Triangle>::iterator Iterator;
27  typedef typename CGAL::AABB_triangle_primitive<K, Iterator> Primitive;
28  typedef typename CGAL::AABB_traits<K, Primitive> AABB_triangle_traits;
29  typedef typename CGAL::AABB_tree<AABB_triangle_traits> Tree;
30  typedef typename Tree::Point_and_primitive_id Point_and_primitive_id;
32  typedef Eigen::Matrix<typename FaceMatrix::Scalar, 3, 1> Face;
33  typedef Eigen::Matrix<typename VertexMatrix::Scalar, 3, 1> Vertex;
34  typedef typename FaceMatrix::Scalar FaceIndex;
35 public:
37  static inline Point_3 tr(Vertex v){ return Point_3(v.x(), v.y(), v.z()); }
38  static inline Vertex tr(Point_3 v){ return Vertex(v.x(), v.y(), v.z()); }
39 private:
40  Tree tree;
41  std::vector<Triangle> triangles;
42 public:
47  template <typename Derived1, typename Derived2>
48  void build( Eigen::MatrixBase<Derived1>& vertices, Eigen::MatrixBase<Derived2>& faces ){
50  for(int fi=0; fi<faces.cols(); fi++){
51  Face f = faces.col(fi);
52  Point_3 v0 = tr( vertices.col(f[0]) );
53  Point_3 v1 = tr( vertices.col(f[1]) );
54  Point_3 v2 = tr( vertices.col(f[2]) );
55  triangles.push_back(Triangle(v0,v1,v2));
56  }
57  // constructs AABB tree
58  tree.rebuild(triangles.begin(),triangles.end());
59  }
60 
67  template <typename Derived1, typename Derived2>
68  void closest_point( const Eigen::MatrixBase<Derived1>& queries, Eigen::MatrixBase<Derived2>& footpoints ){
69  for(int iq=0; iq<queries.cols(); iq++){
70  Point_3 query = tr(queries.col(iq));
71  Point_3 pp = tree.closest_point(query);
72  footpoints.col(iq) = tr(pp);
73  }
74  }
75 
77  template <typename Derived>
78  Eigen::Matrix<Derived, 3, 1> closest_point(Eigen::Matrix<Derived, 3, 1> query){
79  return tr( tree.closest_point(tr(query)) );
80  }
81 
89  template <typename Derived1, typename Derived2, typename Derived3>
90  void closest_point( const Eigen::MatrixBase<Derived1>& queries, Eigen::MatrixBase<Derived2>& footpoints, Eigen::MatrixBase<Derived3>& indexes ){
91  for(int iq=0; iq<queries.cols(); iq++){
92  Point_3 query = tr(queries.col(iq));
93  Point_and_primitive_id pp = tree.closest_point_and_primitive(query);
94  Iterator id = pp.second;
95  std::size_t index_in_vector = std::distance(triangles.begin(),id);
96  assert( triangles[index_in_vector] == *id );
97  footpoints.col(iq) = tr(pp.first);
98  indexes(iq) = index_in_vector;
99  }
100  }
101 
107  template <typename Derived1, typename Derived2, typename Derived3>
108  void barycentric( const Eigen::MatrixBase<Derived1>& footpoints, const Eigen::MatrixBase<Derived2>& indexes, Eigen::MatrixBase<Derived3>& coordinates ){
109  Eigen::Matrix<FT, 3,3> A;
110  for(int iq=0; iq<footpoints.cols(); iq++){
111  const Vertex& footpoint = footpoints.col(iq);
112  FaceIndex fi = indexes(iq);
113  Triangle& triangle = triangles[fi];
114  A.col(0) = tr(triangle[0]);
115  A.col(1) = tr(triangle[1]);
116  A.col(2) = tr(triangle[2]);
117  coordinates.col(iq) = A.inverse() * footpoint;
118  }
119  }
120 };
