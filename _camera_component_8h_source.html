84  void draw(int width, int height) {
85 
86  RenderContext context;
87 
88  glViewport(0, 0, width, height);
89  glClearColor(0.15f, 0.15f, 0.15f, 1);
90  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
91  glEnable(GL_DEPTH_TEST);
92 
93  context.aspect = (float)width / (float)height;
94  context.vfov = vfov;
95  context.near = near_plane;
96  context.far = far_plane;
97  context.eye = get<TransformComponent>().position;
98  context.forward = get<TransformComponent>().forward();
99  context.up = get<TransformComponent>().up();
100 
101  context.update_view();
102  context.update_projection();
103 
104  for (auto &renderable : get_scene().all_of_type<WorldRenderComponent>()) {
105 
106  if (!renderable.visible)
107  continue;
108 
109  auto &transform = renderable.get<TransformComponent>();
110 
111  context.translation = transform.position;
112  context.scale = transform.scale;
113  context.rotation = transform.rotation;
114 
115  context.update_model();
116 
117  renderable.get_renderer().render(context);
118  }
119 
120  GUIRenderEvent event;
121 
122  send_event(event);
123 
124  }
