13 inline Scalar _randn( const Scalar mean, const Scalar stddev ){
14  Scalar u1 = ( static_cast<Scalar> ( std::rand() ) + 1 ) / ( (Scalar)RAND_MAX + 1 );
15  Scalar u2 = ( static_cast<Scalar> ( std::rand() ) + 1 ) / ( (Scalar)RAND_MAX + 1 );
16  assert( -2 * std::log( u1 ) >= 0 );
17  Scalar t1 = std::sqrt( -2 * std::log( u1 ) ) * cos( 2 * M_PI * u2 );
18  return mean + stddev * t1;
19 }
