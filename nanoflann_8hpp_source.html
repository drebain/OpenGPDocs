95  inline void addPoint(DistanceType dist, IndexType index)
96  {
97  CountType i;
98  for (i=count; i>0; --i) {
99 #ifdef NANOFLANN_FIRST_MATCH // If defined and two poins have the same distance, the one with the lowest-index will be returned first.
100  if ( (dists[i-1]>dist) || ((dist==dists[i-1])&&(indices[i-1]>index)) ) {
101 #else
102  if (dists[i-1]>dist) {
103 #endif
104  if (i<capacity) {
105  dists[i] = dists[i-1];
106  indices[i] = indices[i-1];
107  }
108  }
109  else break;
110  }
111  if (i<capacity) {
112  dists[i] = dist;
113  indices[i] = index;
114  }
115  if (count<capacity) count++;
116  }
117 
118  inline DistanceType worstDist() const
119  {
120  return dists[capacity-1];
121  }
122  };

258  inline DistanceType operator()(const T* a, const size_t b_idx, size_t size, DistanceType worst_dist = -1) const
259  {
260  DistanceType result = DistanceType();
261  const T* last = a + size;
262  const T* lastgroup = last - 3;
263  size_t d = 0;
264 
265  /* Process 4 items with each loop for efficiency. */
266  while (a < lastgroup) {
267  const DistanceType diff0 = nanoflann::abs(a[0] - data_source.kdtree_get_pt(b_idx,d++));
268  const DistanceType diff1 = nanoflann::abs(a[1] - data_source.kdtree_get_pt(b_idx,d++));
269  const DistanceType diff2 = nanoflann::abs(a[2] - data_source.kdtree_get_pt(b_idx,d++));
270  const DistanceType diff3 = nanoflann::abs(a[3] - data_source.kdtree_get_pt(b_idx,d++));
271  result += diff0 + diff1 + diff2 + diff3;
272  a += 4;
273  if ((worst_dist>0)&&(result>worst_dist)) {
274  return result;
275  }
276  }
277  /* Process last 0-3 components. Not needed for standard vector lengths. */
278  while (a < last) {
279  result += nanoflann::abs( *a++ - data_source.kdtree_get_pt(b_idx,d++) );
280  }
281  return result;
282  }

329  }

457  class PooledAllocator
458  {
459  /* We maintain memory alignment to word boundaries by requiring that all
460  allocations be in multiples of the machine wordsize. */
461  /* Size of machine word in bytes. Must be power of 2. */
462  /* Minimum number of bytes requested at a time from the system. Must be multiple of WORDSIZE. */
463 
464 
465  size_t remaining; /* Number of bytes left in current block of storage. */
466  void* base; /* Pointer to base of current block of storage. */
467  void* loc; /* Current location in block to next allocate memory. */
468  size_t blocksize;
469 
470  void internal_init()
471  {
472  remaining = 0;
473  base = NULL;
474  usedMemory = 0;
475  wastedMemory = 0;
476  }
477 
478  public:
479  size_t usedMemory;
480  size_t wastedMemory;

485  PooledAllocator(const size_t blocksize_ = BLOCKSIZE){
486  this->blocksize = blocksize_;
487  internal_init();
488  }

498  void free_all()
499  {
500  while (base != NULL) {
501  void *prev = *((void**) base); /* Get pointer to prev block. */
502  ::free(base);
503  base = prev;
504  }
505  internal_init();
506  }

512  void* malloc(const size_t req_size)
513  {
514  /* Round size up to a multiple of wordsize. The following expression
515  only works for WORDSIZE that is a power of 2, by masking last bits of
516  incremented size to zero.
517  */
518  const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
519 
520  /* Check whether a new block must be allocated. Note that the first word
521  of a block is reserved for a pointer to the previous block.
522  */
523  if (size > remaining) {
524 
525  wastedMemory += remaining;
526 
527  /* Allocate new storage. */
528  const size_t blocksize = (size + sizeof(void*) + (WORDSIZE-1) > BLOCKSIZE) ?
529  size + sizeof(void*) + (WORDSIZE-1) : BLOCKSIZE;
530 
531  // use the standard C malloc to allocate memory
532  void* m = ::malloc(blocksize);
533  if (!m) {
534  fprintf(stderr,"Failed to allocate memory.\n");
535  return NULL;
536  }
537 
538  /* Fill first word of new block with pointer to previous block. */
539  ((void**) m)[0] = base;
540  base = m;
541 
542  size_t shift = 0;
543  //int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
544 
545  remaining = blocksize - sizeof(void*) - shift;
546  loc = ((char*)m + sizeof(void*) + shift);
547  }
548  void* rloc = loc;
549  loc = (char*)loc + size;
550  remaining -= size;
551 
552  usedMemory += size;
553 
554  return rloc;
555  }

1271  template <class RESULTSET>
1272  void searchLevel(RESULTSET& result_set, const ElementType* vec, const NodePtr node, DistanceType mindistsq,
1273  distance_vector_t& dists, const float epsError) const
1274  {
1275  /* If this is a leaf node, then do check and return. */
1276  if ((node->child1 == NULL)&&(node->child2 == NULL)) {
1277  //count_leaf += (node->lr.right-node->lr.left); // Removed since was neither used nor returned to the user.
1278  DistanceType worst_dist = result_set.worstDist();
1279  for (IndexType i=node->lr.left; i<node->lr.right; ++i) {
1280  const IndexType index = vind[i];// reorder... : i;
1281  DistanceType dist = distance(vec, index, (DIM>0 ? DIM : dim));
1282  if (dist<worst_dist) {
1283  result_set.addPoint(dist,vind[i]);
1284  }
1285  }
1286  return;
1287  }
1288 
1289  /* Which child branch should be taken first? */
1290  int idx = node->sub.divfeat;
1291  ElementType val = vec[idx];
1292  DistanceType diff1 = val - node->sub.divlow;
1293  DistanceType diff2 = val - node->sub.divhigh;
1294 
1295  NodePtr bestChild;
1296  NodePtr otherChild;
1297  DistanceType cut_dist;
1298  if ((diff1+diff2)<0) {
1299  bestChild = node->child1;
1300  otherChild = node->child2;
1301  cut_dist = distance.accum_dist(val, node->sub.divhigh, idx);
1302  }
1303  else {
1304  bestChild = node->child2;
1305  otherChild = node->child1;
1306  cut_dist = distance.accum_dist( val, node->sub.divlow, idx);
1307  }
1308 
1309  /* Call recursively to search next level down. */
1310  searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError);
1311 
1312  DistanceType dst = dists[idx];
1313  mindistsq = mindistsq + cut_dist - dst;
1314  dists[idx] = cut_dist;
1315  if (mindistsq*epsError<=result_set.worstDist()) {
1316  searchLevel(result_set, vec, otherChild, mindistsq, dists, epsError);
1317  }
1318  dists[idx] = dst;
1319  }
