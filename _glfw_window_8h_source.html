57  glfwGetWindowSize(_window, &w_window, &h_window);

76  GlfwWindow(const std::string& title="", int width=640, int height=480){
77  //--- GLFW initialization
78  {
79  // note: can be called multiple times without problems
80  // docs: "additional calls to glfwInit() after successful initialization but before
81  // termination will return GL_TRUE immediately"
82  if( !glfwInit() )
83  mFatal() << "Failed to initialize GLFW";
84 
85  // Setup
86  glfwSetErrorCallback(glfw_error_callback);
87 
88  // Hint GLFW that we would like a shader-based OpenGL context
89  // docs: "sets hints for the next call to glfwCreateWindow"
90  glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
91  glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 2);
92  glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
93  glfwWindowHint(GLFW_RESIZABLE, GL_TRUE);
94  #ifdef __APPLE__
95  // TODO test if we can use this on Windows and Linux (makes all calls below 3.2 obsolete)
96  glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
97  #endif
98 
101  if( !(_window = glfwCreateWindow(width, height, title.c_str(), nullptr, nullptr)) )
102  mFatal() << "Failed to open OpenGL 3 GLFW window.";
103 
105  if(_enable_verbose_log){
106  int major = glfwGetWindowAttrib(_window, GLFW_CONTEXT_VERSION_MAJOR);
107  int minor = glfwGetWindowAttrib(_window, GLFW_CONTEXT_VERSION_MINOR);
108  int revision = glfwGetWindowAttrib(_window, GLFW_CONTEXT_REVISION);
109  std::cout << "Opened GLFW OpenGL " << major << "." << minor << "." << revision << std::endl;
110  }
111 
112  glfwMakeContextCurrent(_window);
113  if(!glfwGetCurrentContext())
114  mFatal() << "Failed to make GLFW context current.";
115  glfwSetInputMode(_window, GLFW_CURSOR, GLFW_CURSOR_NORMAL); // can be GLFW_CURSOR_HIDDEN
116 
117  // GLEW Initialization (must have created a context)
118  glewExperimental = GL_TRUE;
119  GLenum err = glewInit();
120  if( err != GLEW_OK )
121  mFatal() << "Failed GLEW initialization";
122 
124  while (glGetError() != GL_NO_ERROR) {}
125  }
126 
127  //--- OpenGL globals
128  {
129  glClearColor(1.0f, 1.0f, 1.0f, 0.0f);
130  glClearDepth(1.0f);
131  glEnable(GL_DEPTH_TEST); // Enable depth test
132  glEnable(GL_MULTISAMPLE);
133  // glDisable(GL_CULL_FACE); // back face triangle cull
134 
135  //--- Enable transparency (vec4 color)
136  glEnable (GL_BLEND);
137  glBlendFunc (GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
138  }
139 
140  //--- Hack (manual) for retina display
141  glfwGetFramebufferSize(_window,&_width,&_height);
142  scene.screen_resize(_width,_height);
143 
144  //--- Register the current window with the manager
145  Manager::bind(this);
146  }

167 class Manager{
168  friend class GlfwWindow;
169  typedef std::pair<GLFWwindow*, GlfwWindow*> Pair;
170 
171 protected:
173  static GlfwFpsCounter* fps_counter(){
174  static GlfwFpsCounter fps_counter;
175  return &fps_counter;
176  }
177 
180  static std::map<GLFWwindow*, GlfwWindow*>* active_windows(){
181  static std::map<GLFWwindow*, GlfwWindow*> _active_windows;
182  return &_active_windows;
183  }
184 
185 public:
186  static void bind(GlfwWindow* window){
187  GLFWwindow* _window = window->_window;
188  active_windows()->emplace(_window, window);
189  //--- connect the dispatcher
190  glfwSetKeyCallback(_window, glfw_key_callback);
191  glfwSetMouseButtonCallback(_window, glfw_mouse_press_callback);
192  glfwSetCursorPosCallback(_window, glfw_mouse_move_callback);
193  glfwSetFramebufferSizeCallback(_window, glfw_framebuffer_size_callback);
194  glfwSetScrollCallback(_window, glfw_scroll_callback);
195  }
196  static void unbind(GlfwWindow* window){
197  GLFWwindow* _window = window->_window;
198  active_windows()->erase(_window);
199  glfwSetKeyCallback(_window, nullptr);
200  glfwSetMouseButtonCallback(_window, nullptr);
201  glfwSetCursorPosCallback(_window, nullptr);
202  glfwSetFramebufferSizeCallback(_window, nullptr);
203  glfwSetScrollCallback(_window, nullptr);
204  }
205 
207  static int run(){
208  while(true){
209  // FPS counter
210  fps_counter()->tick();
211 
212  // Empty the event poll to make sure no event is called on deleted window
213  bool _enable_wait_events = false;
214  _enable_wait_events ? glfwWaitEvents() : glfwPollEvents();
215 
216  std::list<GlfwWindow*> todelete;
217  for(Pair win: *active_windows()){
218  if(glfwWindowShouldClose(win.first))
219  todelete.push_back(win.second);
220  else{
221  glfwMakeContextCurrent(win.first);
222  win.second->display();
223  win.second->user_update_fn();
224  glfwSwapBuffers(win.first);
225  }
226  }
227 
228  // Now delete the window from the event subsystem
229  for(GlfwWindow* win: todelete){
230  Manager::unbind(win);
231  glfwDestroyWindow(win->_window);
232  }
233 
234  // Should we terminate the event loop?
235  if(active_windows()->empty()){
236  glfwTerminate();
237  return EXIT_SUCCESS;
238  }
239  }
240  }
242 
244 private:
245  static void glfw_key_callback(GLFWwindow* window, int key, int scancode, int action, int mods){ active_windows()->at(window)->key_callback(key, scancode, action, mods); }
246  static void glfw_mouse_press_callback(GLFWwindow* window, int button, int action, int mods) { active_windows()->at(window)->mouse_press_callback(button, action, mods); }
247  static void glfw_mouse_move_callback(GLFWwindow* window, double xPos, double yPos) { active_windows()->at(window)->mouse_move_callback(xPos, yPos); }
248  static void glfw_framebuffer_size_callback(GLFWwindow* window, int width, int height) { active_windows()->at(window)->framebuffer_size_callback(width, height); }
249  static void glfw_scroll_callback(GLFWwindow* window, double xOffset, double yOffset) { active_windows()->at(window)->scroll_callback(xOffset, yOffset); }
251 };
