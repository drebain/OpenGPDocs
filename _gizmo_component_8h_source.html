31  auto &input = event.camera.get_window().get_input();

33  if (input.get_key(GLFW_KEY_T)) {

39  }
40 
41  auto &transform = get<TransformComponent>();
42 
43  Mat4x4 proj = event.camera.get_projection();
44  Mat4x4 view = event.camera.get_view();
45  Mat4x4 model = transform.get_transformation_matrix();
46  Mat4x4 dmodel = Mat4x4::Zero();
47 
48  ImGuizmo::Manipulate(view.data(), proj.data(), operation, ImGuizmo::LOCAL, model.data(), dmodel.data());
49 
50  switch(operation) {
51  case ImGuizmo::TRANSLATE: transform.apply_translation_matrix(dmodel); break;
52  case ImGuizmo::ROTATE: transform.apply_rotation_matrix(dmodel); break;
53  case ImGuizmo::SCALE: transform.set_scale_matrix(model); break;
54  }

56
