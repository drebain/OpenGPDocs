24 class Global_properties{
26  class Base_global_property{
27  public:
28  const std::type_info& mytype;
29  Base_global_property(const std::type_info& mytype=typeid(void))
30  : mytype( mytype ){}
31  };

34  template <class T>
35  class Global_property : public Base_global_property{
36  public:
37  T value;
38  Global_property() : Base_global_property( typeid(T) ){}
39  };
40 
41 private:
42  typedef std::map<std::string, Base_global_property*> PropertiesMap;
43  PropertiesMap global_props_;
44 
45 public:
46  ~Global_properties(){
47  // std::cout << global_props_.size() << std::endl;
48  for(PropertiesMap::iterator it=global_props_.begin(); it!=global_props_.end(); it++)
49  delete (it->second);
50  }
51 
52 public:
54  template <class T>
55  T& add_property(const std::string& name){
56  if(global_props_.find(name) != global_props_.end())
57  throw std::runtime_error("Attempted to create property with same name");
58  Global_property<T>* prop = new Global_property<T>();
59  global_props_[name] = prop;
60  return prop->value;
61  }
62 
64  template <class T>
65  T& add_property(const std::string& name, const T& initval){
66  T& val = add_property<T>(name);
67  val = initval;
68  return val;
69  }
70 
71  // get a property by its name. returns invalid property if it does not exist.
72  template <class T>
73  T& get_property(const std::string& name){
74  if(global_props_.find(name) == global_props_.end())
75  throw std::runtime_error("Cannot find property");
76  if(global_props_[name]->mytype != typeid(T))
77  throw std::runtime_error("Property of desired type not found");
79  Global_property<T>* prop = static_cast< Global_property<T>* >(global_props_[name]);
80  return prop->value;
81  }
82 };
