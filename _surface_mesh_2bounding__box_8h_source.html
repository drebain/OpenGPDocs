34 inline Box3 bounding_box(const SurfaceMesh& mesh)
35 {
36  auto vpoints = mesh.get_vertex_property<Vec3>("v:point");
37  Box3 bbox;
38  bbox.setNull();
39  for(auto v: mesh.vertices())
40  bbox.extend( vpoints[v] );
41  return bbox;
42 }

45 inline Box3 bbox_cubified(const Box3& box){
46  // TODO: move this function to OpenGP/Types
47  Vec3 centre = box.center();
48  Vec3 d2 = .5 * box.diagonal();
49  // find biggest sizes
50  Scalar s = 0;
51  s = std::max(s, d2.x());
52  s = std::max(s, d2.y());
53  s = std::max(s, d2.z());
54  Vec3 d2new = Vec3(s,s,s);
55  Vec3 bMin = centre-d2new;
56  Vec3 bMax = centre+d2new;
57  return Box3(bMin, bMax);
58 }
