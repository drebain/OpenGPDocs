16 inline Mat4x4 perspective(Scalar fovy, Scalar aspect, Scalar zNear, Scalar zFar) {
17  Eigen::Transform<Scalar,3,Eigen::Projective> tr;
18  tr.matrix().setZero();
19  assert(aspect > 0);
20  assert(zFar > zNear);
21  assert(zNear > 0);
22  Scalar radf = M_PI * fovy / 180.0;
23  Scalar tan_half_fovy = std::tan(radf / 2.0);
24  tr(0,0) = 1.0 / (aspect * tan_half_fovy);
25  tr(1,1) = 1.0 / (tan_half_fovy);
26  tr(2,2) = - (zFar + zNear) / (zFar - zNear);
27  tr(3,2) = - 1.0;
28  tr(2,3) = - (2.0 * zFar * zNear) / (zFar - zNear);
29  return tr.matrix();
30 }

33 template<typename Scalar>
34 inline Mat4x4 ortho( Scalar const& left, Scalar const& right,
35  Scalar const& bottom, Scalar const& top,
36  Scalar const& zNear, Scalar const& zFar )
37 {
38  Eigen::Matrix<Scalar,4,4> mat = Eigen::Matrix<Scalar,4,4>::Identity();
39  mat(0,0) = Scalar(2) / (right - left);
40  mat(1,1) = Scalar(2) / (top - bottom);
41  mat(2,2) = - Scalar(2) / (zFar - zNear);
42  mat(3,0) = - (right + left) / (right - left);
43  mat(3,1) = - (top + bottom) / (top - bottom);
44  mat(3,2) = - (zFar + zNear) / (zFar - zNear);
45  return mat;
46 }

48 inline Mat4x4 scale(Scalar x, Scalar y, Scalar z) {
49  Eigen::Transform<Scalar,3,Eigen::Affine> tr;
50  tr.matrix().setZero();
51  tr(0,0) = x;
52  tr(1,1) = y;
53  tr(2,2) = z;
54  tr(3,3) = 1;
55  return tr.matrix();

70 template<typename Derived>
71 Eigen::Matrix<typename Derived::Scalar,4,4> lookAt(Derived const& eye, Derived const& center, Derived const& up) {
72  typedef Eigen::Matrix<typename Derived::Scalar,4,4> Matrix4;
73  typedef Eigen::Matrix<typename Derived::Scalar,3,1> Vector3;
74  Vector3 f = (center - eye).normalized();
75  Vector3 u = up.normalized();
76  Vector3 s = f.cross(u).normalized();
77  u = s.cross(f);
78  Matrix4 mat = Matrix4::Zero();
79  mat(0,0) = s.x();
80  mat(0,1) = s.y();
81  mat(0,2) = s.z();
82  mat(0,3) = -s.dot(eye);
83  mat(1,0) = u.x();
84  mat(1,1) = u.y();
85  mat(1,2) = u.z();
86  mat(1,3) = -u.dot(eye);
87  mat(2,0) = -f.x();
88  mat(2,1) = -f.y();
89  mat(2,2) = -f.z();
90  mat(2,3) = f.dot(eye);
91  mat.row(3) << 0,0,0,1;
92  return mat;
93 }
