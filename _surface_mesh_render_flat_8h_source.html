28 class SurfaceMeshRenderFlat : public SceneObject{
29 private:
30  SurfaceMesh& mesh;
31  VertexArrayObject vao;
32  ArrayBuffer<Vec3> vertexbuffer;
33  ArrayBuffer<Vec3> normalbuffer;
34  ArrayBuffer<Vec3> barycbuffer;
35 
36 private:
37  const GLchar* vshader = R"GLSL(
38  #version 330 core
39  uniform mat4 M;
40  uniform mat4 MV;
41  uniform mat4 MVP;
42  in vec3 vbaryc; ///< per-vertex barycentric
43  in vec3 vposition; ///< per-vertex position
44  in vec3 vnormal; ///< per-vertex normal
45  out vec3 fnormal; ///< per-fragment normal
46  out vec3 fbaryc; ///< per-fragment barycentric
47  void main(){
48  gl_Position = MVP * vec4(vposition, 1.0);
49  fnormal = normalize( inverse(transpose(mat3(MV))) * vnormal );
50  fbaryc = vbaryc;
51  }
52  )GLSL";
53 
54  const char* fshader = R"GLSL(
55  #version 330 core
56  // uniform vec3 LDIR; ///< TODO: fix me
57  in vec3 fnormal;
58  in vec3 fbaryc;
59  out vec4 fcolor;
60 
61  // http://codeflow.org/entries/2012/aug/02/easy-wireframe-display-with-barycentric-coordinates
62  float edgeFactor(){
63  vec3 d = 1.5 * fwidth(fbaryc);
64  vec3 a3 = smoothstep(vec3(0.0), d, fbaryc);
65  return min(min(a3.x, a3.y), a3.z);
66  }
67 
68  void main(){
70  vec3 LDIR = vec3(0,0,1);
71  vec3 basecolor = vec3(1,0,0);
72  vec3 ldir = normalize(LDIR);
73  float albedo = max( dot( normalize(fnormal), ldir ), 0 );
74  vec4 face_color = vec4(basecolor*albedo, 1);
75 
77  vec4 edge_color = vec4(0,0,0,1);
78  fcolor = mix(edge_color, face_color, edgeFactor());
79  }
80  )GLSL";
81 
82 public:
83  SurfaceMeshRenderFlat(SurfaceMesh& mesh) : mesh(mesh){}
84 
85  void init(){
87  program.add_vshader_from_source(vshader);
88  program.add_fshader_from_source(fshader);
89  program.link();
90 
92  init_data();
93 
95  program.bind();
96  vao.bind();
97  program.set_attribute("vposition", vertexbuffer);
98  program.set_attribute("vnormal", normalbuffer);
99  program.set_attribute("vbaryc", barycbuffer);
100  vao.unbind();
101  program.unbind();
102  }
103 
104  void init_data(){
105  CHECK(mesh.is_triangle_mesh());
106  CHECK(program.is_valid());
107 
108  std::vector<Vec3> v_tri;
109  std::vector<Vec3> n_tri;
110  std::vector<Vec3> b_tri;
111 
112  auto vpoints = mesh.get_vertex_property<Vec3>("v:point");
113  auto fnormals = mesh.get_face_property<Vec3>("f:normal");
114  CHECK(fnormals);
115  CHECK(vpoints);
116 
117  std::vector<Vec3> baryc(3);
118  baryc[0] = Vec3(1,0,0);
119  baryc[1] = Vec3(0,1,0);
120  baryc[2] = Vec3(0,0,1);
121 
123  {
124  for(auto f: mesh.faces()){
125  Vec3 fnormal = fnormals[f];
126  int v_idx = 0;
127  for(auto v: mesh.vertices(f)){
128  v_tri.push_back(vpoints[v]);
130  n_tri.push_back(fnormal);
131  b_tri.push_back(baryc[v_idx++]);
132  }
133  }
134  }
135 
136  vertexbuffer.upload(v_tri);
137  normalbuffer.upload(n_tri);
138  barycbuffer.upload(b_tri);
139  }
140 
141  Box3 bounding_box(){ return OpenGP::bounding_box(mesh); }
142 
143  void display(){
144  program.bind();
145  vao.bind();
146  glDrawArrays(GL_TRIANGLES, 0, mesh.n_faces()*3 /*#verts*/);
147  vao.unbind();
148  program.unbind();
149  }
150 };
